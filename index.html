كيفية ربط الكود بـ FFmpeg.wasm الحقيقي:

يمكنك بسهولة استبدال دالة المحاكاة startProcessing() في ملف الـ Javascript بـ: 

const { createFFmpeg, fetchFile } = FFmpeg;
const ffmpeg = createFFmpeg({ log: true });
await ffmpeg.load();
ffmpeg.FS('writeFile', 'video.mp4', await fetchFile(videoFile));
ffmpeg.FS('writeFile', 'audio.mp3', await fetchFile(audioFile));
await ffmpeg.run('-i', 'video.mp4', '-i', 'audio.mp3', '-c:v', 'copy', '-c:a', 'aac', '-map', '0:v:0', '-map', '1:a:0', 'output.mp4');
const data = ffmpeg.FS('readFile', 'output.mp4');
